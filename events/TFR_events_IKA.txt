    id = IKA.18
    title = IKA.18.t
    desc = IKA.18.d

    is_triggered_only = yes
    option = {
        name = IKA.18.a
        if = {
            limit = {
                balance_of_power = {
                    id = IKA_balance_of_power
                    value < -0.25
                }
            }
            country_event = {
                id = IKA.19
            }
        }
        else_if = {
            limit = {
                balance_of_power = {
                    id = IKA_balance_of_power
                    value > 0.25
                }
            }
            country_event = {
                id = IKA.20
            }
        }
        else = {
            country_event = {
                id = IKA.21
            }
        }
    }
}
